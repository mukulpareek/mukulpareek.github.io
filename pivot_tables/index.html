    Pivot tables

    Specify:
    - Index, ie those columns that appear on the left, equivalent to the 'Rows' in an Excel pivot
    - Values, ie the columns that need the sum or other function performed on them
    - Columns, ie the columns by whose uniquesh the Values above are split into the pivot columns

    You can specify the aggregation functions, which can be multiple (default is np.mean), and also different for different Values cols.

    We start with a random dataframe.

    import pandas as pd
import numpy as np

    data = pd.DataFrame(
    {'state': ['New York', 'Florida', 'Arizona', 'New York', 'Florida', 'Arizona'], 
     'location': ['East', 'East', 'West', 'East', 'East', 'West'],
     'males': list(np.random.randint(100,150,6)),
     'females': list(np.random.randint(100,150,6)),
     'children': list(np.random.randint(100,150,6)),
     'politics': list(np.random.choice(["Red", "Blue"], size=(6), p=[.5, .5]))
    })

    Basic pivot table

    data.pivot_table(index=["state","location"], # Index values on the left, ie the rows
                 values=["males","females"], # DF cols to apply aggfunc to, & show in pivot results
                 columns = ['politics'], # Break the columns down by uniques of these df.cols
                 aggfunc=[np.sum,np.mean], # agg functions to apply, can be multiple. Default=np.mean
                 fill_value=0, # If blanks result, ie NANs, then replace with zero
                 dropna=True, # Drop NAs in the data for the pivot table
                 margins=True) # Provide totals

    # Here we don't specify the columns, just the values
data.pivot_table(index=["state","location"],
                 values=["males","politics"],
                 aggfunc={"politics":[pd.Series.nunique],"males":[np.sum, np.mean]},
                 fill_value=0,
                 dropna=True,
                 margins=False)

    Crosstab

    # Simple crosstab between two columns of a df
pd.crosstab(index = data['state'], columns = data['politics'])
    politics Blue Red
    state
    Arizona 1 1
    Florida 0 2
    New York 0 2 
    # You can make crosstabs more complex with columns and values and aggfunc
pd.crosstab(index = data['state'], 
            columns = data['politics'], 
            values = data['males'], 
            aggfunc = np.sum)
    politics Blue Red
    state
    Arizona 122.0 133.0
    Florida NaN 243.0
    New York NaN 282.0
